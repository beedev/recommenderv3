67 Build base MATCH clause for component type.

96 return query, params

98 def add_compatibility_filters( 

99 self, 

100 query: str, 

101 params: Dict[str, Any], 

102 component_type: str, 

103 selected_components: Dict[str, Any], 

104 node_alias: str = "target", 

105 collect_parent_gins: bool = False 

106 ) -> Tuple[str, Dict[str, Any], Optional[str]]: 

107 """ 

108 Add COMPATIBLE_WITH relationship filters for dependent components. 

109 

110 Supports both single-select (core components) and multi-select (accessories) dependencies. 

111 

112 Args: 

113 query: Existing query string 

114 params: Existing parameters dict 

115 component_type: Type of component being searched 

116 selected_components: Already selected components (ResponseJSON) 

117 node_alias: Alias for the target node 

118 collect_parent_gins: If True, collect parent GINs for attribution (conditional accessories) 

119 

120 Returns: 

121 Tuple of (updated_query, updated_params, parent_alias): 

122 - updated_query: Query with compatibility filters 

123 - updated_params: Parameters with GINs 

124 - parent_alias: Alias for parent node (if collect_parent_gins=True), else None 

125 

126 Example (Single-select - Core Components): 

127 For feeder search with selected power_source: 

128 >>> query = "MATCH (target:Feeder)" 

129 >>> query, params, _ = builder.add_compatibility_filters( 

130 ... query, {}, "feeder", {"PowerSource": selected_ps}, "target" 

131 ... ) 

132 >>> # Adds: MATCH (ps:PowerSource {gin: $ps_gin}) 

133 >>> # MATCH (target)-[:COMPATIBLE_WITH]->(ps) 

134 

135 Example (Multi-select - Accessories): 

136 For feeder_conditional_accessories with 3 selected feeder_accessories: 

137 >>> query = "MATCH (target:Product)" 

138 >>> query, params, parent_alias = builder.add_compatibility_filters( 

139 ... query, {}, "feeder_conditional_accessories", 

140 ... {"FeederAccessories": [acc1, acc2, acc3]}, "target", collect_parent_gins=True 

141 ... ) 

142 >>> # Adds: MATCH (fe_dep:Product) 

143 >>> # WHERE fe_dep.gin IN [$feeder_accessories_gin_0, $feeder_accessories_gin_1, ...] 

144 >>> # MATCH (target)-[:COMPATIBLE_WITH]->(fe_dep) 

145 >>> # Returns parent_alias = "fe_dep" for COLLECT(DISTINCT fe_dep.gin) in caller 

146 """ 

147 config = self.component_config.get(component_type) 

148 requires_compat = config.get("requires_compatibility") if config else None 

149 logger.info(f"🔍 add_compatibility_filters called for {component_type}") 

150 logger.info(f" requires_compatibility: {requires_compat}") 

151 logger.info(f" config exists: {config is not None}") 

152 

153 if not config or not config.get("requires_compatibility"): 

154 logger.info(f" ⏭️ SKIPPING compatibility filters (requires_compatibility=False or no config)") 

155 return query, params, None 

156 

157 dependencies = config.get("dependencies", []) 

158 logger.info(f" dependencies: {dependencies}") 

159 if not dependencies: 

160 return query, params, None 

161 

162 compatibility_clauses = [] 

163 parent_alias = None # Track parent alias for GIN collection 

164 

165 for dep in dependencies: 

166 # Map dependency key to ResponseJSON key 

167 dep_key_map = { 

168 "power_source": "PowerSource", 

169 "feeder": "Feeder", 

170 "cooler": "Cooler", 

171 "feeder_accessories": "FeederAccessories", 

172 "remote_accessories": "RemoteAccessories" 

173 } 

174 response_key = dep_key_map.get(dep) 

175 logger.info(f" Processing dependency: {dep} → {response_key}") 

176 

177 if not response_key: 

178 logger.info(f" ⏭️ No response_key mapping for {dep}") 

179 continue 

180 

181 # Support both dict and Pydantic model access 

182 if hasattr(selected_components, response_key): 

183 selected = getattr(selected_components, response_key) 

184 elif isinstance(selected_components, dict) and response_key in selected_components: 

185 selected = selected_components[response_key] 

186 else: 

187 logger.info(f" ⏭️ {response_key} not found in selected_components") 

188 continue 

189 

190 if not selected: 

191 logger.info(f" ⏭️ {response_key} is None/empty") 

192 continue 

193 

194 logger.info(f" ✅ Found selected component: {response_key} = {selected}") 

195 

196 # Get dependency config 

197 dep_config = self.component_config.get(dep) 

198 if not dep_config: 

199 continue 

200 

201 dep_label = dep_config["neo4j_label"] 

202 dep_alias = f"{dep[:2]}_dep" # e.g., "po_dep" for power_source, "fe_dep" for feeder_accessories 

203 

204 # Check if this is a list (multi-select) or single item 

205 if isinstance(selected, list): 

206 # Multi-select dependency (accessories) 

207 if len(selected) == 0: 

208 continue 

209 

210 # Extract GINs from all selected accessories 

211 gins = [] 

212 for item in selected: 

213 gin = item.get("gin") if isinstance(item, dict) else getattr(item, "gin", None) 

214 if gin: 

215 gins.append(gin) 

216 

217 if not gins: 

218 continue 

219 

220 # Create parameter list for WHERE IN clause 

221 param_names = [] 

222 for idx, gin in enumerate(gins): 

223 param_name = f"{dep}_gin_{idx}" 

224 params[param_name] = gin 

225 param_names.append(f"${param_name}") 

226 

227 # Add MATCH clause with WHERE IN for multiple GINs 

228 compatibility_clauses.append( 

229 f"MATCH ({dep_alias}:{dep_label})" 

230 ) 

231 compatibility_clauses.append( 

232 f"WHERE {dep_alias}.gin IN [{', '.join(param_names)}]" 

233 ) 

234 

235 # Add COMPATIBLE_WITH relationship - target compatible with ANY parent 

236 compatibility_clauses.append( 

237 f"MATCH ({node_alias})-[r:COMPATIBLE_WITH]->({dep_alias})" 

238 ) 

239 

240 # Track parent alias for GIN collection 

241 if collect_parent_gins: 

242 parent_alias = dep_alias 

243 

244 logger.info( 

245 f"Added multi-product compatibility filter: {component_type} -> {dep} " 

246 f"({len(gins)} products: {gins})" 

247 ) 

248 

249 else: 

250 # Single-select dependency (core component) 

251 gin = selected.get("gin") if isinstance(selected, dict) else getattr(selected, "gin", None) 

252 if not gin: 

253 continue 

254 

255 param_name = f"{dep}_gin" 

256 params[param_name] = gin 

257 

258 # Add MATCH clause for dependency 

259 compatibility_clauses.append( 

260 f"MATCH ({dep_alias}:{dep_label} {{gin: ${param_name}}})" 

261 ) 

262 

263 # Add COMPATIBLE_WITH relationship 

264 compatibility_clauses.append( 

265 f"MATCH ({node_alias})-[r:COMPATIBLE_WITH]->({dep_alias})" 

266 ) 

267 

268 logger.info(f"Added single-product compatibility filter: {component_type} -> {dep} (GIN: {gin})") 

269 

270 if compatibility_clauses: 

271 query = "\n".join([query] + compatibility_clauses) 

272 

273 return query, params, parent_alias

275 def add_search_term_filters( 

276 self, 

277 query: str, 

278 params: Dict[str, Any], 

279 search_terms_dict: Dict[str, Any], 

280 node_alias: str 

281 ) -> Tuple[str, Dict[str, Any]]: 

282 """ 

283 Add dynamic WHERE/AND search-term filters to Cypher query. 

284 

285 USER REQUIREMENT (from previous implementation): 

286 "If product_name has value - match on item_name using CONTAINS. 

287 For all features, use CONTAINS on clean_description or attribute_ruleset." 

288 

289 Args: 

290 query: Existing query string 

291 params: Existing parameters dict 

292 search_terms_dict: Dict with "product_name" and "feature_terms" keys 

293 node_alias: Neo4j node alias 

294 

295 Returns: 

296 Tuple of (updated_query, updated_params) 

297 """ 

298 product_name = search_terms_dict.get("product_name") 

299 feature_terms = search_terms_dict.get("feature_terms", []) 

300 

301 if not product_name and not feature_terms: 

302 return query, params 

303 

304 conditions = [] 

305 

306 # 1. If product_name exists: Match on item_name using CONTAINS (space-insensitive) 

307 if product_name: 

308 param_name = "product_name_filter" 

309 conditions.append( 

310 f"replace(toLower({node_alias}.item_name), ' ', '') CONTAINS replace(toLower(${param_name}), ' ', '')" 

311 ) 

312 params[param_name] = product_name 

313 logger.info(f"Added space-insensitive item_name filter for product_name: '{product_name}'") 

314 

315 # 2. For all feature terms: Match on clean_description or attribute_ruleset using CONTAINS 

316 if feature_terms: 

317 for idx, term in enumerate(feature_terms): 

318 param_name = f"feature_{idx}" 

319 conditions.append( 

320 f"(toLower({node_alias}.clean_description) CONTAINS toLower(${param_name}) " 

321 f"OR toLower({node_alias}.attribute_ruleset) CONTAINS toLower(${param_name}))" 

322 ) 

323 params[param_name] = term 

324 

325 logger.info(f"Added {len(feature_terms)} feature filters on clean_description/attribute_ruleset") 

326 

327 if conditions: 

328 query_stripped = query.strip() 

329 # Check if the last clause is a MATCH - if so, use WHERE 

330 # Otherwise, if WHERE exists earlier, use AND 

331 last_line = query_stripped.split('\n')[-1].strip().upper() 

332 if last_line.startswith('MATCH'): 

333 query += "\nWHERE (" + " OR ".join(conditions) + ")" 

334 elif "WHERE" in query_stripped.upper(): 

335 query += " AND (" + " OR ".join(conditions) + ")" 

336 else: 

337 query += " WHERE (" + " OR ".join(conditions) + ")" 

338 

339 return query, params

393 def build_lucene_query( 

394 self, 

395 component_type: str, 

396 user_message: str, 

397 node_alias: str = "p" 

398 ) -> Tuple[str, Dict[str, Any]]: 

399 """ 

400 Build Lucene full-text search query for component type. 

401 

402 Uses OLD PRODUCT_SEARCH PATTERN: 

403 1. Remove stop words ("I", "need", "want", etc.) 

404 2. Normalize text (units, spaces) 

405 3. Build UNION query: 

406 - Query 1: Normalized text (better precision) 

407 - Query 2: Stop-words-removed text (better recall) 

408 4. Combine results with DISTINCT and order by score DESC 

409 

410 Args: 

411 component_type: Type of component 

412 user_message: User's search message 

413 node_alias: Neo4j node alias 

414 

415 Returns: 

416 Tuple of (query_string, params_dict) 

417 

418 Example: 

419 >>> query, params = builder.build_lucene_query( 

420 ... "power_source", "I need a 500A MIG welder", "p" 

421 ... ) 

422 >>> # Returns UNION query searching both: 

423 >>> # - Normalized: "500A MIG welder" 

424 >>> # - Original: "500A MIG welder" (stop words removed) 

425 """ 

426 config = self.component_config.get(component_type) 

427 if not config: 

428 raise ValueError(f"Unknown component type: {component_type}") 

429 

430 if not config.get("lucene_enabled"): 

431 raise ValueError(f"Lucene search not enabled for {component_type}") 

432 

433 neo4j_label = config["neo4j_label"] 

434 category = config["category"] 

435 

436 # 🔧 ENHANCEMENT (Nov 15, 2024): Multi-step query processing from old product_search.py 

437 # Step 1: Remove stop words 

438 stopwords_removed = self._remove_stopwords(user_message) 

439 logger.info(f"Step 1 - Stopwords removed: '{user_message}' → '{stopwords_removed}'") 

440 

441 # Step 2: Normalize text (units, spaces, etc.) 

442 normalized_text = self._normalize_search_text(stopwords_removed) 

443 logger.info(f"Step 2 - Normalized: '{stopwords_removed}' → '{normalized_text}'") 

444 

445 # Step 3: Escape Lucene special characters 

446 escaped_normalized = self._escape_lucene_special_chars(normalized_text) 

447 escaped_stopwords = self._escape_lucene_special_chars(stopwords_removed) 

448 

449 # Step 4: Build UNION query if normalized differs from stopwords-removed 

450 # This provides better recall by searching both versions 

451 if normalized_text != stopwords_removed: 

452 logger.info(f"Building UNION query: normalized vs stopwords-removed") 

453 

454 # Neo4j 5.x requires UNION queries to be wrapped in CALL subquery 

455 # with each part having its own RETURN clause 

456 query = f""" 

457 CALL {{ 

458 CALL db.index.fulltext.queryNodes('productIndex', $normalized_text) 

459 YIELD node AS {node_alias}, score 

460 WITH * 

461 WHERE {node_alias}:{neo4j_label} 

462 """ 

463 if category: 

464 query += f" AND {node_alias}.category = $category" 

465 

466 query += f""" 

467 RETURN {node_alias}, score 

468 UNION 

469 CALL db.index.fulltext.queryNodes('productIndex', $stopwords_text) 

470 YIELD node AS {node_alias}, score 

471 WITH * 

472 WHERE {node_alias}:{neo4j_label} 

473 """ 

474 if category: 

475 query += f" AND {node_alias}.category = $category" 

476 

477 query += f""" 

478 RETURN {node_alias}, score 

479 }} 

480 """ 

481 

482 params = { 

483 "normalized_text": escaped_normalized, 

484 "stopwords_text": escaped_stopwords, 

485 "category": category 

486 } 

487 else: 

488 # Single query when no normalization happened 

489 logger.info(f"Building single query: '{stopwords_removed}'") 

490 

491 query = f""" 

492 CALL db.index.fulltext.queryNodes('productIndex', $search_text) 

493 YIELD node AS {node_alias}, score 

494 WITH * 

495 WHERE {node_alias}:{neo4j_label} 

496 """ 

497 if category: 

498 query += f" AND {node_alias}.category = $category" 

499 

500 params = { 

501 "search_text": escaped_stopwords, 

502 "category": category 

503 } 

504 

505 logger.debug(f"Built Lucene query for {component_type}") 

506 return query, params

542 def extract_and_normalize_keywords(self, text: str) -> List[str]: 

543 """ 

544 Extract keywords from text and normalize measurements. 

545 

546 Reuses logic from product_search.py for consistency. 

547 

548 Examples: 

549 - "Renegade ES300" → ["Renegade", "ES300"] 

550 - "5m interconnector" → ["5m", "5 m", "5.0 m", "5.0m", "interconnector"] 

551 - "70mm Gas-Cooled" → ["70mm", "70 mm", "70.0 mm", "70.0mm", "Gas-Cooled"] 

552 """ 

553 keywords = [] 

554 tokens = re.split(r'[\s,;]+', text) 

555 

556 for token in tokens: 

557 if not token.strip(): 

558 continue 

559 

560 # Check if token contains measurement 

561 if re.search(r'\d+\.?\d*\s*(m|mm|cm|km)\b', token, flags=re.IGNORECASE): 

562 normalized = self._expand_measurement_terms(token) 

563 keywords.extend(normalized) 

564 else: 

565 keywords.append(token.strip()) 

566 

567 return keywords

569 def _normalize_parameter_value(self, value: str, component_type: str) -> List[str]: 

570 """ 

571 Normalize parameter values using comprehensive parameter_normalizations.json mappings. 

572 

573 This replaces the old _expand_measurement_terms() method with a more robust 

574 approach that handles: 

575 - Cable lengths (2m, 5m, 10m, etc.) 

576 - Current ratings (300A, 400A, 500A, etc.) 

577 - Voltages (230V, 400V, 480V, etc.) 

578 - Wire diameters (0.8mm, 1.0mm, 1.2mm, etc.) 

579 - Cross sections (70mm², 95mm²) 

580 - Duty cycles (60%, 100%) 

581 - Cooling capacity (3L/min, 5L/min) 

582 - Swan neck angles (45°, 90°) 

583 

584 Args: 

585 value: Parameter value to normalize (e.g., "5m", "500A", "70mm") 

586 component_type: Component type (e.g., "power_source", "feeder") 

587 

588 Returns: 

589 List of normalized variants for fuzzy matching 

590 

591 Examples: 

592 >>> _normalize_parameter_value("5m", "interconnector") 

593 ["5m", "5 m", "5.0m", "5 meter", "5 meters", "5-m"] 

594 

595 >>> _normalize_parameter_value("500A", "power_source") 

596 ["500A", "500 A", "500amp", "500 ampere", "500 amps", "500-A"] 

597 

598 >>> _normalize_parameter_value("70mm", "interconnector") 

599 ["70mm", "70mm²", "70mm2", "70 mm²", "70 mm2", "70mm squared", "70-mm²"] 

600 """ 

601 if not value or not self.normalizations: 

602 return [value.strip()] 

603 

604 value_normalized = value.strip().lower() 

605 

606 # Check each normalization type relevant to this component 

607 applicable_params = self.component_parameter_mapping.get(component_type, []) 

608 

609 for param_type in applicable_params: 

610 if param_type not in self.normalizations: 

611 continue 

612 

613 mappings = self.normalizations[param_type].get("mappings", {}) 

614 

615 # Search for matching canonical value 

616 for canonical_value, variants in mappings.items(): 

617 # Check if input matches any variant (case-insensitive) 

618 for variant in variants: 

619 if variant.lower() == value_normalized: 

620 logger.debug( 

621 f"Normalized '{value}' → {len(variants)} variants for {param_type}" 

622 ) 

623 return variants # Return ALL variants for this canonical value 

624 

625 # Fallback: Try basic pattern matching for unmapped values 

626 return self._fallback_normalization(value)

628 def _fallback_normalization(self, value: str) -> List[str]: 

629 """ 

630 Fallback normalization for values not in parameter_normalizations.json. 

631 

632 Applies basic pattern matching for common measurement types. 

633 

634 Args: 

635 value: Parameter value 

636 

637 Returns: 

638 List of normalized variants 

639 """ 

640 variants = [value.strip()] 

641 

642 # Pattern 1: Length measurements (m, mm, cm, km) 

643 length_pattern = r'\b(\d+\.?\d*)\s*(m|mm|cm|km)\b' 

644 match = re.search(length_pattern, value, flags=re.IGNORECASE) 

645 if match: 

646 number = match.group(1) 

647 unit = match.group(2).lower() 

648 variants.append(f"{number} {unit}") 

649 if '.' not in number: 

650 variants.append(f"{number}.0 {unit}") 

651 variants.append(f"{number}.0{unit}") 

652 return variants 

653 

654 # Pattern 2: Current ratings (A, amp, ampere) 

655 current_pattern = r'\b(\d+\.?\d*)\s*(?:A|amp|ampere)s?\b' 

656 match = re.search(current_pattern, value, flags=re.IGNORECASE) 

657 if match: 

658 number = match.group(1) 

659 variants.extend([f"{number}A", f"{number} A", f"{number}amp"]) 

660 return variants 

661 

662 # Pattern 3: Voltage (V, volt) 

663 voltage_pattern = r'\b(\d+\.?\d*)\s*(?:V|volt)s?\b' 

664 match = re.search(voltage_pattern, value, flags=re.IGNORECASE) 

665 if match: 

666 number = match.group(1) 

667 variants.extend([f"{number}V", f"{number} V", f"{number}volt"]) 

668 return variants 

669 

670 # Pattern 4: Percentages (%, percent) 

671 percent_pattern = r'\b(\d+\.?\d*)\s*(?:%|percent)\b' 

672 match = re.search(percent_pattern, value, flags=re.IGNORECASE) 

673 if match: 

674 number = match.group(1) 

675 variants.extend([f"{number}%", f"{number} %", f"{number}percent"]) 

676 return variants

680 def build_search_terms_from_component( 

681 self, 

682 component_dict: Dict[str, Optional[str]], 

683 component_type: str 

684 ) -> Dict[str, Any]: 

685 """ 

686 Build search terms from ALL extracted features (universal approach). 

687 

688 USER REQUIREMENT (UNIVERSAL): 

689 "This approach works for all products - extract all features, normalize if needed, 

690 use CONTAINS against clean_description or attribute_ruleset. 

691 If product_name exists, also match on item_name." 

692 

693 ENHANCED (Nov 15, 2024): Split comma-separated values for better matching 

694 - Process field: "MIG (GMAW), MAG, MMA/Stick" → ["MIG (GMAW)", "MAG", "MMA/Stick"] 

695 - Each individual term searched separately for ANY match (OR logic) 

696 

697 Returns dict with: 

698 - "product_name": Original product_name for item_name CONTAINS matching (if exists) 

699 - "feature_terms": All extracted feature values (normalized) for description/attribute matching 

700 """ 

701 result = { 

702 "product_name": None, 

703 "feature_terms": [] 

704 } 

705 

706 if not component_dict or not isinstance(component_dict, dict): 

707 return result 

708 

709 for key, value in component_dict.items(): 

710 if not value or not isinstance(value, str) or not value.strip(): 

711 continue 

712 

713 value_stripped = value.strip() 

714 

715 if key == "product_name": 

716 # Check for GIN (skip if GIN detected) 

717 if re.match(r'^\d{10}$', value_stripped): 

718 logger.info(f"GIN detected in product_name field: {value_stripped}") 

719 continue 

720 

721 result["product_name"] = value_stripped 

722 # Also add keywords from product_name to feature_terms 

723 keywords = self.extract_and_normalize_keywords(value_stripped) 

724 result["feature_terms"].extend(keywords) 

725 else: 

726 # 🔧 FIX (Nov 15, 2024): Split comma-separated values for better matching 

727 # Root cause: "MIG (GMAW), MAG, MMA/Stick, DC TIG (GTAW)" searched as single string 

728 # Solution: Split by comma and search each process individually 

729 if "," in value_stripped: 

730 logger.info(f"Splitting comma-separated value for '{key}': {value_stripped}") 

731 # Split by comma and process each term individually 

732 individual_terms = [term.strip() for term in value_stripped.split(",")] 

733 for term in individual_terms: 

734 if term: # Skip empty strings 

735 normalized_terms = self._normalize_parameter_value(term, component_type) 

736 result["feature_terms"].extend(normalized_terms) 

737 logger.info(f" → Split into {len(individual_terms)} individual terms") 

738 else: 

739 # Single value - normalize as before 

740 normalized_terms = self._normalize_parameter_value(value_stripped, component_type) 

741 result["feature_terms"].extend(normalized_terms) 

742 

743 logger.info( 

744 f"Extracted search terms: product_name='{result['product_name']}', " 

745 f"{len(result['feature_terms'])} feature terms: {result['feature_terms']}" 

746 ) 

747 

748 return result

750 def _escape_lucene_special_chars(self, text: str) -> str: 

751 """ 

752 Escape Lucene special characters in search text. 

753 

754 Lucene query parser has special characters that need escaping: 

755 + - && || ! ( ) { } [ ] ^ " ~ * ? : \ / 

756 

757 Nov 15, 2024: Added to fix parsing errors when user queries contain 

758 special characters like "MIG/MAG" or "MMA (Stick)". 

759 

760 Args: 

761 text: Raw search text 

762 

763 Returns: 

764 Text with Lucene special characters escaped 

765 

766 Example: 

767 >>> self._escape_lucene_special_chars("MIG/MAG welding") 

768 "MIG\\/MAG welding" 

769 >>> self._escape_lucene_special_chars("MMA (Stick) welding") 

770 "MMA \\(Stick\\) welding" 

771 """ 

772 # Lucene special characters that need escaping with backslash 

773 special_chars = ['+', '-', '&', '|', '!', '(', ')', '{', '}', '[', ']', 

774 '^', '"', '~', '*', '?', ':', '\\', '/'] 

775 

776 escaped_text = text 

777 for char in special_chars: 

778 # Escape each special character with backslash 

779 escaped_text = escaped_text.replace(char, f'\\{char}') 

780 

781 return escaped_text

783 def _remove_stopwords(self, text: str) -> str: 

784 """ 

785 Remove stop words from search text. 

786 

787 Stop words are common English words that don't contribute to search relevance 

788 (e.g., "I", "need", "want", "for", "a", "the", etc.). 

789 

790 Nov 15, 2024: Imported from old product_search.py pattern for better Lucene matching. 

791 

792 Args: 

793 text: Raw search text 

794 

795 Returns: 

796 Text with stop words removed 

797 

798 Example: 

799 >>> self._remove_stopwords("I need a 500A MIG welder") 

800 "500A MIG welder" 

801 >>> self._remove_stopwords("I want to find a machine for welding") 

802 "machine welding" 

803 """ 

804 if not text or not isinstance(text, str): 

805 return text 

806 

807 # Convert to lowercase and normalize whitespace 

808 clean = text.lower().strip() 

809 clean = re.sub(r'\s+', ' ', clean) 

810 

811 # Split into words 

812 words = clean.split() 

813 

814 # Remove stop words 

815 keywords = [ 

816 w for w in words 

817 if w not in STOP_WORDS and len(w) > 1 

818 ] 

819 

820 # If all words were stop words, keep words with length > 1 

821 if not keywords and words: 

822 keywords = [w for w in words if len(w) > 1] 

823 

824 result = ' '.join(keywords) 

825 return result

827 def _normalize_search_text(self, text: str) -> str: 

828 """ 

829 Normalize measurement units in search text. 

830 

831 Converts various unit formats to standardized forms that match database content. 

832 This preprocessing step improves Lucene search accuracy by ensuring consistent 

833 terminology between user queries and product descriptions. 

834 

835 Nov 15, 2024: Imported from old product_search.py pattern. 

836 

837 Args: 

838 text: Search text (typically after stop words removed) 

839 

840 Returns: 

841 Normalized search text with standardized units 

842 

843 Examples: 

844 >>> self._normalize_search_text("500 Amps MIG welder") 

845 "500 A MIG welder" 

846 >>> self._normalize_search_text("380 Volts 30mm wire") 

847 "380 V 30 mm wire" 

848 >>> self._normalize_search_text("18mm wire 3ph power") 

849 "18 mm wire 3 phase power" 

850 

851 Normalization Rules: 

852 - Amperage: Amps/Ampere/Ampères → A 

853 - Voltage: Volts/Volt/Voltios → V 

854 - Length: meters/metres → m, millimeters/millimetres → mm 

855 - Power: Watts → W, kilowatts → kW 

856 - Pressure: Bar/BAR → bar 

857 - Flow: l/min, liters/minute → l/minute 

858 - Phase: 3ph, 1-phase → 3 phase, 1 phase 

859 """ 

860 normalized = text 

861 

862 # Rule 1: Add space between numbers and units if missing 

863 # "500A" → "500 A", "30mm" → "30 mm" 

864 normalized = re.sub(r'(\d+)([A-Za-z]+)', r'\1 \2', normalized) 

865 

866 # Rule 2: Amperage - normalize to "A" 

867 # "500 Amps" → "500 A", "500 Ampere" → "500 A", "500 Ampères" → "500 A" 

868 normalized = re.sub( 

869 r'(\d+)\s*(Amps?|Amperes?|Ampères?)\b', 

870 r'\1 A', 

871 normalized, 

872 flags=re.IGNORECASE 

873 ) 

874 

875 # Rule 3: Voltage - normalize to "V" 

876 # "380 Volts" → "380 V", "460 Volt" → "460 V", "380 Voltios" → "380 V" 

877 normalized = re.sub( 

878 r'(\d+)\s*(Volts?|Voltios?)\b', 

879 r'\1 V', 

880 normalized, 

881 flags=re.IGNORECASE 

882 ) 

883 

884 # Rule 4: Power (Watts) - normalize to "W" 

885 # "500 Watts" → "500 W", "500 watt" → "500 W" 

886 normalized = re.sub( 

887 r'(\d+)\s*Watts?\b', 

888 r'\1 W', 

889 normalized, 

890 flags=re.IGNORECASE 

891 ) 

892 

893 # Rule 5: Power (Kilowatts) - normalize to "kW" 

894 # "4 kilowatt" → "4 kW", "4 kilowatts" → "4 kW" 

895 normalized = re.sub( 

896 r'(\d+)\s*kilowatts?\b', 

897 r'\1 kW', 

898 normalized, 

899 flags=re.IGNORECASE 

900 ) 

901 

902 # Rule 6: Length (meters) - normalize to "m" 

903 # "15 meters" → "15 m", "5 metre" → "5 m" 

904 normalized = re.sub( 

905 r'(\d+)\s*(meters?|metres?)\b', 

906 r'\1 m', 

907 normalized, 

908 flags=re.IGNORECASE 

909 ) 

910 

911 # Rule 7: Length (millimeters) - normalize to "mm" 

912 # "30 millimeters" → "30 mm", "1.2 millimetres" → "1.2 mm" 

913 normalized = re.sub( 

914 r'(\d+(?:\.\d+)?)\s*(millimeters?|millimetres?)\b', 

915 r'\1 mm', 

916 normalized, 

917 flags=re.IGNORECASE 

918 ) 

919 

920 # Rule 8: Length (inches) - normalize to "inch" 

921 # '32"' → "32 inch", "32 inches" → "32 inch" 

922 normalized = re.sub( 

923 r'(\d+)\s*(?:inches?|")\b', 

924 r'\1 inch', 

925 normalized, 

926 flags=re.IGNORECASE 

927 ) 

928 

929 # Rule 9: Pressure (bar) - normalize to lowercase "bar" 

930 # "5 Bar" → "5 bar", "5 BAR" → "5 bar" 

931 normalized = re.sub( 

932 r'(\d+)\s*bar\b', 

933 r'\1 bar', 

934 normalized, 

935 flags=re.IGNORECASE 

936 ) 

937 

938 # Rule 10: Flow rate - normalize to "l/minute" 

939 # "7 l/min" → "7 l/minute", "7 liters/minute" → "7 l/minute" 

940 normalized = re.sub( 

941 r'(\d+)\s*(?:l/min|liters?/min(?:ute)?|lpm)\b', 

942 r'\1 l/minute', 

943 normalized, 

944 flags=re.IGNORECASE 

945 ) 

946 

947 # Rule 11: Phase - normalize to "phase" 

948 # "3ph" → "3 phase", "1-phase" → "1 phase" 

949 normalized = re.sub( 

950 r'(\d+)[-\s]*ph\b', 

951 r'\1 phase', 

952 normalized, 

953 flags=re.IGNORECASE 

954 ) 

955 

956 # Rule 12: Duty cycle percent - preserve as-is 

957 # Already in correct format: "60%", "@60%", "at 60%" 

958 # No normalization needed for duty cycle 

959 

960 return normalized.strip()
