70 self,

72 state: str, 

73 products: List[Dict[str, Any]] = None,

83 """ 

84 Unified response generation router - maps message types to appropriate methods.

86 Args: 

87 message_type: Type of message ("selection", "auto_selection", "search_results", "error", "finalize", "skip") 

88 state: Current configurator state 

89 products: List of products (for search_results) 

90 selected_product: Selected product (for selection/auto_selection)

92 custom_message: Optional custom message to prepend 

93 error_type: Error type (for error messages) 

94 details: Error details (for error messages) 

95 response_json: Response JSON (for finalize) 

96 zero_results_message: Message to display when no products found 

97 compatibility_skip_message: Message explaining compatibility validation yielded no results 

98 

99 Returns: 

100 Generated message string 

101 """ 

102 # Convert Pydantic model to dict if needed (similar to parameter_extractor.py fix) 

103 if selected_product and hasattr(selected_product, 'dict'): 

104 selected_product = selected_product.dict() 

105 

106 if message_type in ["selection", "auto_selection"]: 

107 # Use custom message if provided, otherwise generate confirmation 

108 if custom_message: 

109 return custom_message 

110 elif selected_product: 

111 component_type = selected_product.get("category", "Component") 

112 return self.generate_selection_confirmation( 

113 component_type, 

114 selected_product.get("name", ""), 

115 selected_product.get("gin", "") 

116 ) 

117 else: 

118 return "Selection confirmed." 

119 

120 elif message_type == "search_results": 

121 # Prepend compatibility skip message if provided 

122 prefix = f"{compatibility_skip_message}\n\n" if compatibility_skip_message else "" 

123 

124 # Generate search results message from product list 

125 if products and len(products) > 0: 

126 # Format product list (simplified version) 

127 component_name = self._get_component_name(state) 

128 message = f"Here are the {component_name} options:\n\n" 

129 

130 # List top 5 products 

131 for idx, product in enumerate(products[:5], 1): 

132 message += f"{idx}. **{product.get('name', 'Unknown')}** (GIN: {product.get('gin', 'N/A')})\n" 

133 

134 message += f"\n✅ Please select a {component_name} or say 'skip' if not needed." 

135 return prefix + message 

136 elif zero_results_message: 

137 return prefix + zero_results_message 

138 else: 

139 state_prompt = await self.generate_state_prompt(state, response_json or {}, language) 

140 return prefix + state_prompt 

141 

142 elif message_type == "error": 

143 # Generate error message 

144 return self.generate_error_message(error_type or "unknown", details) 

145 

146 elif message_type == "finalize":

158 component_key = state.replace("_selection", "") 

159 return self.generate_skip_confirmation(component_key, component_key)

161 else: 

162 # Unknown message type - return generic message

165 async def generate_qa_response( 

166 self, 

167 question: str, 

168 context: Dict[str, Any], 

169 language: str = "en", 

170 ) -> str: 

171 """ 

172 Generate LLM-powered answer to user questions. 

173 🔒 THREE-LAYER ANTI-HALLUCINATION PROTECTION: 

174 - Layer 1: Competitor query blocking 

175 - Layer 2: Vague query normalization with ESAB fallbacks 

176 - Layer 3: ESAB-only system prompt enforcement 

177 

178 Args: 

179 question: User's question 

180 context: Current session context with keys: 

181 - current_state: Current configuration state 

182 - response_json: Selected components 

183 - master_parameters: User requirements 

184 language: Target language for response 

185 

186 Returns: 

187 Natural language answer maintaining conversational flow 

188 """ 

189 try: 

190 # 🔒 LAYER 1 — Competitor guard (prevent brand hallucination) 

191 if self._is_competitor_query(question): 

192 logger.info("🚫 Competitor query blocked for domain safety") 

193 return ( 

194 "Sorry, I can only recommend ESAB welding equipment and accessories " 

195 "compatible with your current setup." 

196 ) 

197 

198 # 🔒 LAYER 2 — Normalize vague "best/good/suggest" questions 

199 # Prevents hallucination by providing concrete ESAB examples 

200 lower_q = question.lower() 

201 if any(keyword in lower_q for keyword in ["best", "good", "suggest", "recommend"]): 

202 # Optional quick ESAB-based fallback from Neo4j 

203 try: 

204 from ..neo4j.product_search import Neo4jProductSearch 

205 search = Neo4jProductSearch("bolt://localhost:7687", "neo4j", "test") 

206 products = await search._simple_neo4j_search("PowerSource", ["aristo"], []) 

207 if products: 

208 top = ", ".join(p.name for p in products[:3]) 

209 return f"ESAB offers excellent power sources such as {top}." 

210 except Exception: 

211 pass 

212 

213 # Fallback to hardcoded ESAB examples 

214 return ( 

215 "ESAB offers several high-performance power sources such as " 

216 "Aristo 500ix, Warrior 500i, and Renegade ES300i." 

217 ) 

218 

219 # 🔒 LAYER 3 — Build contextual prompt with ESAB-only enforcement 

220 prompt = self._build_qa_prompt(question, context) 

221 

222 # Get response from model with ESAB-only system prompt 

223 answer = await self._call_llm_for_qa(prompt, language) 

224 

225 if not answer.endswith(('.', '!', '?')): 

226 answer += '.' 

227 

228 logger.info(f"✅ Q&A response generated for question: {question[:50]}...") 

229 return answer 

230 

231 except Exception as e: 

232 logger.error(f"❌ Q&A generation failed: {e}", exc_info=True) 

233 return self._fallback_qa_response(question)

235 async def _call_llm_for_qa(self, prompt: str, language: str) -> str: 

236 """ 

237 Call OpenAI for Q&A responses. 

238 🔒 USES ESAB-ONLY SYSTEM PROMPT to prevent hallucination 

239 """ 

240 # Use ESAB-restricted system prompt 

241 system_prompt = self.ESAB_ONLY_SYSTEM_PROMPT 

242 

243 if language != "en": 

244 system_prompt += f"\nRespond in {language}." 

245 

246 response = await self.openai_client.chat.completions.create( 

247 model="gpt-4o-mini", 

248 messages=[ 

249 {"role": "system", "content": system_prompt}, 

250 {"role": "user", "content": prompt} 

251 ], 

252 temperature=0.4, # Lower temperature for more factual responses 

253 max_tokens=300, 

254 timeout=10.0 

255 ) 

256 

257 return response.choices[0].message.content.strip()

296 def _format_selections(self, response_json: Dict[str, Any]) -> str: 

297 """Format selected components for context display""" 

298 if hasattr(response_json, 'dict'): 

299 response_json = response_json.dict()

304 return "None selected yet" 

305 

306 selections = [] 

307 component_map = { 

308 "PowerSource": "Power Source", 

309 "Feeder": "Feeder", 

310 "Cooler": "Cooler", 

311 "Interconnector": "Interconnector", 

312 "Torch": "Torch", 

313 "Accessories": "Accessories", 

314 "PowerSourceAccessories": "PowerSource Accessories", 

315 "FeederAccessories": "Feeder Accessories", 

316 "FeederConditionalAccessories": "Feeder Conditional Accessories",

321 "Connectivity": "Connectivity Modules",

323 } 

324 

325 for comp_key, data in response_json.items(): 

326 if not data: 

327 continue 

328
