    Orchestrator Architecture - StateByStateOrchestrator¶

    File: src/backend/app/services/orchestrator/state_orchestrator.py

    The StateByStateOrchestrator is the central coordinator of the 3-agent system, managing the S1â†’SN dynamic state machine workflow.

    Overview¶

    The orchestrator is a thin coordination layer that delegates all state-specific logic to modular processors via the StateProcessorRegistry. It focuses solely on:

    • State transitions between processors
    • Special command handling (skip, done, finalize)
    • Session management
    • Compound request coordination

    Architecture Pattern: Registry + Delegation Pattern (~800-1000 lines vs previous 3,851 lines)

    def _load_applicability_for_powersource(self, power_source_gin: str) -> ComponentApplicability:
    """
    Load component applicability from configuration for a specific PowerSource.

    Args:
        power_source_gin: GIN of the selected PowerSource

    Returns:
        ComponentApplicability with Y/N/O rules for each component
    """
    power_sources = self.powersource_applicability_config.get("power_sources", {})
    ps_config = power_sources.get(power_source_gin)

    if ps_config:
        # Found config for this specific PowerSource
        applicability_data = ps_config.get("applicability", {})
        logger.info(f"   Loaded applicability for {ps_config.get('name', power_source_gin)}: {applicability_data}")
        return ComponentApplicability(**applicability_data)
    else:
        # PowerSource not in config - use default policy
        default_policy = self.powersource_applicability_config.get("default_policy", {})
        applicability_data = default_policy.get("applicability", {})
        logger.warning(
            f"   PowerSource {power_source_gin} not in config - using default applicability: {applicability_data}"
        )
        return ComponentApplicability(**applicability_data)

    Configuration: Loads from component_applicability.json which defines Y/N/O flags for each component based on selected PowerSource.

    @traceable(name="orchestrator_process_message", run_type="chain")
async def process_message(
    self,
    conversation_state: ConversationState,
    user_message: str,
    last_shown_products: Optional[List[Dict]] = None,
) -> Dict[str, Any]:
    """
    Process user message in current state using modular processors.

    Flow:
    1. Check for special commands (skip, done, finalize)
    2. Extract parameters via ParameterExtractor (Agent 1)
    3. Detect compound requests
    4. Delegate to processor for current state
    5. Handle state transition
    6. Generate response via MessageGenerator (Agent 3)

    Args:
        conversation_state: Current session state
        user_message: User's input message
        last_shown_products: Previously shown products (for selection detection)

    Returns:
        Dict with response, state, products, selection status
    """
    language = conversation_state.language or "en"
    logger.info(f"ðŸŽ¯ Processing message in state: {conversation_state.current_state}")

    try:
        # STEP 1: Handle special commands first
        if self._is_special_command(user_message):
            return await self._handle_special_command(
                user_message, conversation_state, language
            )

        # STEP 2: Extract parameters using Agent 1 (LLM)
        master_parameters = await self.parameter_extractor.extract_parameters(
            user_message=user_message,
            current_state=conversation_state.current_state,
            master_parameters=conversation_state.master_parameters,
        )

        # Update conversation state with extracted parameters
        conversation_state.master_parameters = master_parameters

        # Log extracted parameters for debugging
        import json
        master_params_dict = master_parameters.model_dump() if hasattr(master_parameters, 'model_dump') else master_parameters
        non_empty_params = {
            k: v for k, v in master_params_dict.items()
            if k != "last_updated" and v and v != {} and v != []
        }
        if non_empty_params:
            logger.info(f"ðŸ” LLM EXTRACTED PARAMETERS:\n{json.dumps(non_empty_params, indent=2)}")
        else:
            logger.info("ðŸ” LLM EXTRACTED: No parameters extracted from user message")

        # STEP 3: Check for compound request (multiple components specified)
        if self._is_compound_request(master_parameters, conversation_state):
            return await self._handle_compound_request(
                master_parameters, conversation_state, language
            )

        # STEP 4: Delegate to processor for current state
        return await self._process_single_state(
            user_message, master_parameters, conversation_state, language
        )

    except Exception as e:
        logger.error(f"âŒ Error in process_message: {e}", exc_info=True)
        return self._generate_error_response(str(e), conversation_state, language)

    Key Features: - LangSmith Tracing: @traceable decorator for observability - Special Commands: Handles skip, done, finalize commands - Agent 1 Integration: Calls ParameterExtractor.extract_parameters() for LLM-based intent extraction - Compound Request Detection: Multi-component requests like "Aristo 500ix with RobustFeed U6" - Processor Delegation: Routes to appropriate state processor for search and validation

    @traceable(name="orchestrator_select_product", run_type="chain")
async def select_product(
    self,
    product_gin: str,
    product_data: Dict[str, Any],
    conversation_state: ConversationState,
    language: str = "en",
) -> Dict[str, Any]:
    """
    Handle explicit product selection using modular processors.

    Flow:
    1. Get processor for current state
    2. Create SelectedProduct from product data
    3. Add to response_json
    4. Check if proactive display enabled for next state
    5. Generate response with selection confirmation

    Args:
        product_gin: Selected product GIN
        product_data: Product details
        conversation_state: Current session state
        language: ISO 639-1 language code

    Returns:
        Dict with response, updated state, next products
    """
    logger.info(f"âœ… Product selected: {product_gin} in state {conversation_state.current_state}")

    try:
        # Guard: Reject selections in FINALIZE state
        if str(conversation_state.current_state) == "finalize" or conversation_state.current_state == ConfiguratorState.FINALIZE:
            logger.warning(f"âŒ Cannot select products in FINALIZE state")
            return {
                "response": "Configuration is complete. Cannot add more products.",
                "current_state": conversation_state.current_state,
                "products": [],
                "can_finalize": True
            }

        # Get processor for current state
        processor = self.registry.get_processor(conversation_state.current_state)
        if not processor:
            raise ValueError(f"No processor found for state: {conversation_state.current_state}")

        # Create selected product
        selected_product = SelectedProduct(
            gin=product_gin,
            name=product_data.get("name", "Unknown"),
            category=product_data.get("category", ""),
            description=product_data.get("description", ""),
        )

        # Add to response_json based on component type
        component_key = processor.component_type
        field_name = config_service.get_response_json_field_name(component_key)

        if processor.is_multi_select():
            # Multi-select: Add to list (e.g., Accessories)
            current_list = getattr(conversation_state.response_json, field_name, [])
            if not isinstance(current_list, list):
                current_list = []
            current_list.append(selected_product)
            setattr(conversation_state.response_json, field_name, current_list)
        else:
            # Single-select: Replace existing
            setattr(conversation_state.response_json, field_name, selected_product)

        # ... handle state transition and response generation

    Key Features: - FINALIZE Guard: Prevents selection after configuration is complete - Multi-Select Support: Handles accessories (multiple) vs single components - Dynamic Field Mapping: Maps component_key to ResponseJSON field names - State Transition: Automatically advances to next applicable state

    def _is_compound_request(self, master_parameters: Any, conversation_state: ConversationState) -> bool:
    """
    Detect if user specified multiple components in one message.

    Example: "Aristo 500ix with RobustFeed U6"
    """
    # Check how many components have specifications
    specified_count = sum(1 for component_params in master_params if component_params)
    return specified_count > 1 or (specified_count == 1 and not at_powersource_state)

        return True, None

        self, user_message, master_parameters, selected_components, limit
    ):
        """Generate proactive preview for this state"""
        if not self.proactive_display:
            return None
        # ... preview generation logic

    def get_next_state(self, conversation_state, selection_made):
    """Determine next state after power source selection"""

    if not selection_made:
        # Error case: No selection made (shouldn't happen - power source is mandatory)
        return "power_source_selection"

    # Get applicability from response_json
    applicability = conversation_state.response_json.applicability.model_dump()

    # Check if Feeder is applicable
    if applicability.get("Feeder") in ["mandatory", "optional", "Y"]:
        return "feeder_selection"

    # Feeder not applicable, check Cooler
    if applicability.get("Cooler") in ["mandatory", "optional", "Y"]:
        return "cooler_selection"

    # Continue through all primary components...
    # Eventually defaults to accessories or finalize

    def _score(self, p: Dict[str, Any], context: Optional[Dict[str, Any]] = None) -> tuple:
    name = (p.get("name") or "").lower()
    is_default = p.get("is_default", False)
    q = (context or {}).get("query") or ""
    q = q.lower().strip()

    exact_hit = (1 if q and q in name else 0)

    # Lower tuple sorts earlier; invert booleans
    return (
        0 if is_default else 1,  # Tier 1: Prefer defaults
        0 if exact_hit else 1,   # Tier 2: Prefer exact name contains
        name                     # Tier 3: Alphabetical
    )

    Sorting Priority:

    1. Tier 1: Default Products (is_default=True)
    2. Products flagged as recommended/default by ESAB
    3. Score: 0 (highest priority)

    4. Use case: Promote most popular or recommended products

    5. Tier 2: Exact Match (query in product name)

    6. Exact phrase match in product name
    7. Score: 0 if match, 1 if no match

    8. Use case: "Aristo 500ix" query matches "Aristo 500ix MIG Welder"

    9. Tier 3: Alphabetical (stable sort)

    10. Deterministic ordering by product name
    11. Use case: Consistent results across requests

    def _build_graph(self) -> StateGraph:
    """Build LangGraph workflow"""

    workflow = StateGraph(ConfiguratorGraphState)

    # Add nodes
    workflow.add_node("extract_parameters", self.extract_parameters_node)
    workflow.add_node("search_products", self.search_products_node)
    workflow.add_node("generate_response", self.generate_response_node)

    workflow.add_edge("search_products", "generate_response")
    workflow.add_edge("generate_response", "determine_next_state")
    workflow.add_edge("determine_next_state", END)

    return workflow

    async def process_message(self, session_id, user_message, current_state):
    """Process message through LangGraph workflow"""

    # Update state with new message
    current_state["user_message"] = user_message
    current_state["messages"].append({
        "role": "user",
        "content": user_message,
        "timestamp": datetime.utcnow().isoformat()
    })

    # Create config for checkpointing
    config = {"configurable": {"thread_id": session_id}}

    # Invoke graph with checkpointing
    result = await self.app.ainvoke(current_state, config)

    # Checkpoint automatically saved by LangGraph
    logger.info(f"Checkpoint #{result.get('checkpoint_count', 0)} saved")

    return result

    Benefits: - Session Persistence: State saved automatically after each node - Resume Capability: Resume from last checkpoint on reconnection - Crash Recovery: Recover state after server restart - Debugging: Inspect graph state at each checkpoint

    Why Not Used in Production?¶

    The main orchestrator (StateByStateOrchestrator) is preferred because:

    1. Simpler: Direct function calls vs graph nodes
    2. More Flexible: Dynamic state transitions based on applicability
    3. Better Performance: No checkpointing overhead
    4. Easier Debugging: Standard Python async/await patterns
    5. Less Dependencies: No LangGraph or Redis checkpointing required

    LangGraph is retained for: - Future experimentation with multi-agent workflows - Alternative architecture research - Checkpointing capabilities if needed - LangSmith-native integration patterns
