    def _safe_get_gin(self, response_json: Dict[str, Any], component_key: str) -> Optional[str]:
    """
    Safely extract GIN from response_json component.
    Handles cases where component might be:
    - A dict with 'gin' key
    - The string "skipped" (skip tracking feature)
    - None
    """
    component = response_json.get(component_key)
    if component is None:
        return None
    if component == "skipped":
        return None  # Treat skipped as no component selected
    if isinstance(component, dict):
        return component.get("gin")
    return None  # Fallback for unexpected types

    response_json: Dict[str, Any],
    limit: int = 3
) -> SearchResults:
    """Search for compatible Torches based on PowerSource, Feeder, and Cooler

    ✨ INTEGRATED COOLER SUPPORT:
    If PowerSource has integrated cooler, cooler compatibility is not required.
    Only PowerSource + Feeder compatibility is checked.
    """

    # Use safe GIN extraction for skip tracking compatibility
    power_source_gin = self._safe_get_gin(response_json, "PowerSource")
    feeder_gin = self._safe_get_gin(response_json, "Feeder")
    cooler_gin = self._safe_get_gin(response_json, "Cooler")

    # ✨ Check for integrated cooler in PowerSource
    power_source_data = response_json.get("PowerSource", {})
    integrated_cooler = False
    if isinstance(power_source_data, dict):
        integrated_cooler = power_source_data.get("integrated_cooler", False) or \
                           power_source_data.get("has_integrated_cooler", False)

    # Track which relationships we have
    has_feeder = bool(feeder_gin)
    # ✨ Cooler is considered present if explicitly selected AND NOT integrated in PowerSource
    has_cooler = bool(cooler_gin) and not integrated_cooler

    # WHERE conditions for missing feeder/cooler
    # ✨ INTEGRATED COOLER: If PowerSource has integrated cooler, has_cooler = False,
    #    so cooler compatibility check is automatically skipped
    where_conditions = []
    if not has_feeder:
        where_conditions.append("NOT EXISTS { MATCH (:Product {category: 'Feeder'})-[:COMPATIBLE_WITH]->(target) }")
    if not has_cooler:
        # Note: This will be skipped if integrated_cooler = True (because has_cooler = False)
        where_conditions.append("NOT EXISTS { MATCH (:Product {category: 'Cooler'})-[:COMPATIBLE_WITH]->(target) }")

    def _build_finalize_prompt(
    self,
    response_json: Dict[str, Any],
    state_config: Dict[str, Any]
) -> str:
    """
    Build finalize prompt with JSON summary using configuration
    Enhanced to include all accessory categories
    Uses finalize_header and finalize_footer from state_prompts.json
    ✨ SKIP TRACKING: Includes skipped components with status
    """
    import json

    # Build clean JSON structure with GIN, name, description for selected items
    # and category/status for skipped items
    clean_config = {}

    # Core components - Handle three states: None (omit), "skipped" (show status), dict (show details)
    for component_type in ["PowerSource", "Feeder", "Cooler", "Interconnector", "Torch"]:
        component_data = response_json.get(component_type)

        if component_data is None:
            # Not applicable (never shown to user) - OMIT from JSON
            continue
        elif component_data == "skipped":
            # Explicitly skipped by user - SHOW WITH STATUS
            clean_config[component_type] = {
                "category": component_type,
                "status": "skipped"
            }
        elif isinstance(component_data, dict):
            # Selected product - SHOW FULL DETAILS
            clean_config[component_type] = {
                "gin": component_data.get("gin"),
                "name": component_data.get("name"),
                "description": component_data.get("description")
            }

    # Accessory categories (all lists)
    accessory_categories = [
        "PowerSourceAccessories",
        "FeederAccessories",
        "FeederConditionalAccessories",
        "InterconnectorAccessories",
        "Remotes",
        "RemoteAccessories",
        "RemoteConditionalAccessories",
        "Connectivity",
        "FeederWears",
        "Accessories"  # Legacy
    ]

    for category in accessory_categories:
        category_data = response_json.get(category)

        if category_data == "skipped":
            # Explicitly skipped by user - SHOW WITH STATUS
            clean_config[category] = {
                "category": category,
                "status": "skipped"
            }
        elif category_data and isinstance(category_data, list) and len(category_data) > 0:
            # Selected products - SHOW FULL DETAILS
            clean_config[category] = [
                {
                    "gin": item.get("gin"),
                    "name": item.get("name"),
                    "description": item.get("description")
                }
                for item in category_data
            ]

    # Format as pretty JSON
    json_str = json.dumps(clean_config, indent=2)

    # Get header and footer from config (fallback to defaults)
    header = state_config.get("finalize_header", "📋 **Final Configuration:**")
    footer = state_config.get("finalize_footer", "\n\n✨ Your configuration is ready!")

    # Build complete prompt
    prompt = f"{header}\n\n```json\n{json_str}\n```{footer}"

    return prompt

    Integrated Cooler Logic¶

    Problem Statement¶

    Some PowerSource units have built-in (integrated) coolers. When a user selects such a PowerSource: - They should NOT be asked to select a separate external Cooler - Downstream components (Interconnector, Torch) should NOT require cooler compatibility - Cooler MATCH and WHERE clauses in Cypher queries should be skipped

    Solution: Formula-Based Approach¶

    Key Formula: 

    1
    has_cooler = bool(cooler_gin) and not integrated_cooler

    How It Works:

    Scenario cooler_gin integrated_cooler has_cooler Result
    No cooler selected, no integrated cooler None False False Cooler required
    Cooler selected, no integrated cooler "0123..." False True Cooler present
    No cooler selected, has integrated cooler None True False Cooler NOT required ✅
    Cooler selected, has integrated cooler "0123..." True False Cooler NOT required ✅

    Result: When integrated_cooler = True, has_cooler is ALWAYS False, automatically skipping cooler compatibility checks.

    Skip Tracking Preservation¶

    What Is Skip Tracking?¶

    Skip tracking distinguishes between three states for each component:

    1. None: Not applicable (component was never shown to user, skipped by applicability rules)
    2. "skipped": Explicitly skipped by user (user declined optional component)
    3. SelectedProduct: User selected a product

    Why It Matters¶

    • User Intent: Distinguishes "not needed" from "not applicable"
    • Analytics: Track which components users decline vs. which are never offered
    • Finalize Summary: Show user what they skipped explicitly
    • Business Intelligence: Understand user preferences and drop-off points
