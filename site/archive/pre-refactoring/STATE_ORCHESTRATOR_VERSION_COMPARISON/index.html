    async def _route_to_state_processor(
    self,
    conversation_state: ConversationState,
    state: ConfiguratorState
) -> Optional[Dict[str, Any]]:
    """Route to the appropriate state processor"""

    state_processors = {
        ConfiguratorState.POWER_SOURCE_SELECTION: self._handle_power_source_selection,
        ConfiguratorState.FEEDER_SELECTION: self._handle_feeder_selection,
        ConfiguratorState.COOLER_SELECTION: self._handle_cooler_selection,
        # ... all other states
    }

    processor = state_processors.get(state)
    if processor:
        return await processor(conversation_state, user_message="")
    return None

    Benefits: Better code organization, enables auto-progression through states

    def _find_next_applicable_state(self, conversation_state: ConversationState) -> ConfiguratorState:
    """
    Find next state where component is applicable and not yet selected.

    CRITICAL: Always returns the FIRST unselected required component in sequence order.
    This ensures we never skip required components like Feeder.
    """

    # 1. Get applicability from response_json
    applicability = self._get_current_applicability(conversation_state)

    # 2. Define state order with numeric indices
    state_order = [
        (0, ConfiguratorState.POWER_SOURCE_SELECTION, "PowerSource"),
        (1, ConfiguratorState.FEEDER_SELECTION, "Feeder"),
        (2, ConfiguratorState.COOLER_SELECTION, "Cooler"),
        (3, ConfiguratorState.INTERCONNECTOR_SELECTION, "Interconnector"),
        (4, ConfiguratorState.TORCH_SELECTION, "Torch"),
        (5, ConfiguratorState.POWERSOURCE_ACCESSORIES_SELECTION, "PowerSourceAccessories"),
        # ... 12 total states
    ]

    # 3. Get current state index
    current_index = -1
    for idx, st, _ in state_order:
        if st == conversation_state.current_state:
            current_index = idx
            break

    # 4. Find FIRST applicable and unselected component AFTER current state
    for idx, next_state, component in state_order:
        if idx <= current_index:
            continue

        # Check applicability
        component_applicability_value = applicability.get(component)

        # Skip if not applicable
        if component_applicability_value in ["N", "not_applicable"]:
            continue

        # Special handling for integrated cooler
        if component == "Cooler" and component_applicability_value == "integrated_cooler":
            continue

        # Check if already selected
        selected = getattr(conversation_state.response_json, component, None)
        if not selected or (isinstance(selected, list) and len(selected) == 0):
            # Found first unselected applicable component
            return next_state

    # All components done - proceed to finalize
    return ConfiguratorState.FINALIZE

    Key Features Preserved: 1. ‚úÖ Sequential state order enforcement 2. ‚úÖ Applicability-based state skipping (N, not_applicable) 3. ‚úÖ Integrated cooler special handling 4. ‚úÖ Numeric index-based state comparison 5. ‚úÖ FIRST unselected component logic (never skip required components) 6. ‚úÖ Accessory base component checking 7. ‚úÖ Automatic progression to FINALIZE when complete
