async def search_power_source_with_fallback(
    self,
    master_params_dict: Dict[str, Any],
    user_message: Optional[str] = None,
    use_lucene: bool = False,
    limit: int = 10
) -> SearchResults:
    """
    Unified search method with optional Lucene support

    Args:
        master_params_dict: Extracted parameters
        user_message: Raw user message for Lucene search
        use_lucene: Whether to attempt Lucene search
        limit: Max results to return

    Returns:
        SearchResults from Lucene or traditional search
    """
    if use_lucene and user_message:
        try:
            # Try Lucene first
            results = await self.search_power_source_lucene(
                user_message=user_message,
                limit=limit,
                min_score=0.5
            )
            if results and results.products:
                logger.info(f"‚úÖ Lucene search returned {len(results.products)} results")
                return results
            else:
                logger.info("‚ö†Ô∏è Lucene returned no results, falling back to traditional search")
        except Exception as e:
            logger.warning(f"‚ö†Ô∏è Lucene search failed: {e}, falling back to traditional search")

    # Traditional search (fallback or default)
    return await self.search_power_source(master_params_dict, limit=limit)

    # File: app/services/neo4j/product_search.py

async def search_power_source_smart(
    self,
    master_params_dict: Dict[str, Any],
    user_message: Optional[str] = None,
    limit: int = 10
) -> SearchResults:
    """
    Intelligent PowerSource search with automatic Lucene/traditional selection

    Reads configuration from search_config.json:
    - If lucene_search.enabled=true AND user_message provided ‚Üí Try Lucene
    - Falls back to traditional search if Lucene fails or returns no results
    - Always returns valid SearchResults
    """
    from ..config.configuration_service import get_config_service

    # Load Lucene configuration
    try:
        config = get_config_service().get_config("search_config")
        lucene_config = config.get("lucene_search", {})
        lucene_enabled = lucene_config.get("enabled", False)
        lucene_min_score = lucene_config.get("min_score", 0.5)
    except Exception as e:
        logger.warning(f"Failed to load search config: {e}, using traditional search")
        lucene_enabled = False

    # Attempt Lucene if enabled and user message available
    if lucene_enabled and user_message:
        try:
            logger.info("üîç Attempting Lucene full-text search for PowerSource")
            results = await self.search_power_source_lucene(
                user_message=user_message,
                limit=limit,
                min_score=lucene_min_score
            )

            if results and results.products:
                logger.info(f"‚úÖ Lucene search successful: {len(results.products)} results")
                return results
            else:
                logger.info("‚ÑπÔ∏è Lucene returned no results, falling back to traditional search")
        except Exception as e:
            logger.warning(f"‚ö†Ô∏è Lucene search failed: {e}, falling back to traditional search")

    # Traditional parameter-based search (default or fallback)
    logger.info("üìä Using traditional parameter-based search for PowerSource")
    return await self.search_power_source(master_params_dict, limit=limit)

    üö® Risk Mitigation¶

    Risk 1: Lucene Index Missing/Broken¶

    Mitigation: Automatic fallback to traditional search Detection: Log warning, monitor error rates Recovery: Rebuild index with create_lucene_index.py

    Risk 2: Performance Degradation¶

    Mitigation: Set timeout on Lucene queries (5s max) Detection: Monitor P95 latency Recovery: Disable Lucene via config

    Risk 3: Incorrect Results¶

    Mitigation: Start with high min_score (0.7+), gradually lower Detection: User feedback, manual testing Recovery: Adjust min_score or disable Lucene
