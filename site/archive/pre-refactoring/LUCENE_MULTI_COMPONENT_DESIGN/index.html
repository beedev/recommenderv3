    async def _lucene_search(
    self,
    component_type: str,
    user_message: str,
    limit: int = 10,
    offset: int = 0,
    min_score: float = 0.5,
    compatibility_filter: Optional[Dict[str, str]] = None
) -> SearchResults:
    """
    Generic Lucene full-text search for any component type.

    Args:
        component_type: Component type key (power_source, feeder, cooler, torch, remote)
        user_message: Raw user message for full-text search
        limit: Max results to return
        offset: Pagination offset
        min_score: Minimum Lucene relevance score
        compatibility_filter: Optional compatibility constraints
            Example: {"power_source_gin": "0446200880", "feeder_gin": "0460520880"}

    Returns:
        SearchResults with scored products
    """
    from ..config.configuration_service import get_config_service

    # Load component-specific configuration
    try:
        config = get_config_service().load_config("search_config")
        lucene_config = config.get("lucene_search", {})
        components_config = lucene_config.get("components", {})
        component_config = components_config.get(component_type, {})

        neo4j_label = component_config.get("neo4j_label", component_type.capitalize())
        neo4j_category = component_config.get("neo4j_category", component_type.capitalize())
        score_threshold_percent = component_config.get(
            "score_threshold_percent",
            lucene_config.get("score_threshold_percent", 25)
        )
        append_score = lucene_config.get("append_score_to_name", True)
        score_decimals = lucene_config.get("score_decimal_places", 1)

        logger.info(f"ğŸ” Lucene search for {component_type}: '{user_message[:100]}...'")
        logger.info(f"Config: label={neo4j_label}, category={neo4j_category}, threshold={score_threshold_percent}%")

    except Exception as e:
        logger.error(f"Failed to load Lucene config for {component_type}: {e}")
        return SearchResults(products=[], total_count=0, filters_applied={"error": str(e)})

    try:
        # Build compatibility WHERE clause if needed
        compatibility_where = ""
        if compatibility_filter:
            conditions = []
            for key, value in compatibility_filter.items():
                conditions.append(f"EXISTS((node)-[:COMPATIBLE_WITH]->(:Node {{gin: '{value}'}}))")
            if conditions:
                compatibility_where = f"AND ({' AND '.join(conditions)})"

        # Generic Lucene query
        query = f"""
        CALL db.index.fulltext.queryNodes("productIndex", $user_message)
        YIELD node, score
        WHERE node.category = $category
          AND score >= $min_score
          {compatibility_where}
        RETURN DISTINCT
            node.gin as gin,
            node.item_name as name,
            node.category as category,
            node.clean_description as description,
            node.attributes_ruleset as specifications_json,
            node as specifications,
            score
        ORDER BY score DESC
        """

        params = {
            "user_message": user_message,
            "category": neo4j_category,
            "min_score": min_score
        }

        async with self.driver.session() as session:
            result = await session.run(query, params)
            records = await result.data()

            # Collect products with scores
            products_with_scores = []
            for record in records:
                specs = record.get("specifications", {})
                if hasattr(specs, "__dict__"):
                    specs = dict(specs)
                specs = self._clean_neo4j_types(specs)

                products_with_scores.append({
                    "product": ProductResult(
                        gin=record["gin"],
                        name=record["name"],
                        category=record["category"],
                        description=record.get("description"),
                        specifications=specs
                    ),
                    "score": record["score"]
                })

            logger.info(f"ğŸ“Š Initial Lucene results for {component_type}: {len(products_with_scores)} products")

            # Apply score threshold filtering (same logic as PowerSource)
            if products_with_scores:
                top_score = products_with_scores[0]["score"]
                threshold_multiplier = 1 - (score_threshold_percent / 100)
                min_threshold = top_score * threshold_multiplier

                logger.info(f"ğŸ¯ Score filtering: top={top_score:.3f}, threshold={score_threshold_percent}%, min={min_threshold:.3f}")

                filtered_products = [
                    item for item in products_with_scores
                    if item["score"] >= min_threshold
                ]

                logger.info(f"âœ‚ï¸ After filtering: {len(filtered_products)} products")

                # Append scores if enabled
                products = []
                for item in filtered_products:
                    product = item["product"]
                    score = item["score"]

                    if append_score:
                        score_display = score * 100
                        product.name = f"{product.name} (Score: {score_display:.{score_decimals}f})"

                    products.append(product)
            else:
                products = []
                logger.info(f"â„¹ï¸ No Lucene results for {component_type}")

            # Apply pagination
            total = len(products)
            paginated = products[offset:offset + limit]
            has_more = (offset + limit) < total

            logger.info(f"âœ… Returning {len(paginated)}/{total} {component_type} products")

            return SearchResults(
                products=paginated,
                total_count=total,
                filters_applied={
                    "search_method": "lucene_fulltext",
                    "component_type": component_type,
                    "user_message": user_message[:100],
                    "compatibility_validated": bool(compatibility_filter)
                },
                compatibility_validated=bool(compatibility_filter),
                offset=offset,
                limit=limit,
                has_more=has_more
            )

    except Exception as e:
        logger.error(f"âŒ Lucene search failed for {component_type}: {e}")
        return SearchResults(
            products=[],
            total_count=0,
            filters_applied={"search_method": "lucene_fulltext", "error": str(e)}
        )

    async def _search_component_smart(
    self,
    component_type: str,
    master_parameters: Dict[str, Any],
    response_json: Optional[Dict[str, Any]] = None,
    user_message: Optional[str] = None,
    limit: int = 10,
    offset: int = 0
) -> SearchResults:
    """
    Generic intelligent search routing for any component.

    Automatically chooses between:
    1. Lucene full-text search (if enabled and user_message provided)
    2. Traditional parameter-based search (fallback)

    Args:
        component_type: Component type (power_source, feeder, cooler, torch, remote)
        master_parameters: Extracted parameters from MasterParameterJSON
        response_json: Selected components (for compatibility validation)
        user_message: Raw user message for Lucene search (optional)
        limit: Max results
        offset: Pagination offset

    Returns:
        SearchResults from either Lucene or traditional search
    """
    from ..config.configuration_service import get_config_service

    # Load configuration
    try:
        config = get_config_service().load_config("search_config")
        lucene_config = config.get("lucene_search", {})
        components_config = lucene_config.get("components", {})
        component_config = components_config.get(component_type, {})

        lucene_enabled = lucene_config.get("enabled", False)
        component_enabled = component_config.get("enabled", False)
        min_score = component_config.get("min_score", 0.5)
    except Exception as e:
        logger.warning(f"Failed to load config for {component_type}: {e}")
        lucene_enabled = False
        component_enabled = False

    # Attempt Lucene if enabled
    if lucene_enabled and component_enabled and user_message:
        try:
            logger.info(f"ğŸ” Attempting Lucene search for {component_type}")

            # Build compatibility filter for dependent components
            compatibility_filter = None
            if response_json and component_type != "power_source":
                compatibility_filter = {}
                if response_json.get("PowerSource"):
                    compatibility_filter["power_source_gin"] = response_json["PowerSource"]["gin"]
                # Add more compatibility filters as needed

            results = await self._lucene_search(
                component_type=component_type,
                user_message=user_message,
                limit=limit,
                offset=offset,
                min_score=min_score,
                compatibility_filter=compatibility_filter
            )

            if results and results.products:
                logger.info(f"âœ… Lucene success for {component_type}: {len(results.products)} results")
                return results
            else:
                logger.info(f"â„¹ï¸ Lucene returned 0 results for {component_type}, falling back")

        except Exception as e:
            logger.warning(f"âš ï¸ Lucene failed for {component_type}: {e}, falling back")

    # Fallback to traditional search
    logger.info(f"ğŸ“Š Using traditional search for {component_type}")

    # Route to existing component-specific traditional methods
    if component_type == "power_source":
        return await self.search_power_source(master_parameters, limit, offset)
    elif component_type == "feeder":
        return await self.search_feeder(master_parameters, response_json, limit, offset)
    elif component_type == "cooler":
        return await self.search_cooler(master_parameters, response_json, limit, offset)
    elif component_type == "torch":
        return await self.search_torch(master_parameters, response_json, limit, offset)
    elif component_type == "remote":
        return await self.search_remotes(master_parameters, response_json, limit, offset)
    else:
        logger.error(f"âŒ Unknown component type: {component_type}")
        return SearchResults(products=[], total_count=0, filters_applied={"error": "Unknown component"})

    Benefits of This Design¶

    1. DRY Principle (Don't Repeat Yourself)¶

    • âœ… One generic Lucene method for all components
    • âœ… One generic smart routing method for all components
    • âœ… Component-specific logic centralized in configuration

    2. Maintainability¶

    • âœ… Fix bugs once, works for all components
    • âœ… Update score filtering logic once, applies everywhere
    • âœ… Easy to understand: Generic core + thin facades

    3. Extensibility¶

    • âœ… Add new component in 3 steps:
    • Add config entry (5 lines)
    • Add facade method (15 lines)
    • Create Neo4j index (1 Cypher query)
    • âœ… No need to duplicate 150+ lines per component

    4. Performance¶

    • âœ… Same Lucene performance as component-specific
    • âœ… No overhead from generic methods (async calls are inlined)

    5. Backward Compatibility¶

    • âœ… Existing traditional search methods unchanged
    • âœ… Can roll out component by component
    • âœ… Easy to disable Lucene per component via config
