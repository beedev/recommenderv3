async def _execute_parallel(self, strategies, ...):
    tasks = [
        asyncio.wait_for(
            strategy.search(...),
            timeout=self.timeout  # âœ… 30 second max per strategy

        for strategy in strategies

async def save_session(self, conversation_state):
    session_data = conversation_state.dict()
