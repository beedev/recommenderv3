    Generic _lucene_search() Architecture¶

    How It Works: 1. Takes compatibility_filter dict with any component GINs 2. Dynamically builds Cypher WHERE clause with EXISTS checks 3. Example for interconnector with PowerSource + Feeder + Cooler:

    Strengths: - Generic and reusable across all component types - Supports multi-component compatibility (AND logic) - Clean separation of concerns

    Weaknesses: - No support for NOT EXISTS (negative filtering) - No support for conditional logic (if/then rules) - No applicability awareness

    async def search_torch_lucene(
    self,
    user_message: str,
    feeder_gin: Optional[str] = None,  # Make optional
    cooler_gin: Optional[str] = None,  # Add cooler
    integrated_cooler: bool = False,  # Add integrated cooler flag
    limit: int = 10,
    offset: int = 0,
    min_score: float = 0.5
) -> SearchResults:
    compatibility_filter = {}
    negative_filter = {}

    # Positive compatibility with selected components
    if feeder_gin:
        compatibility_filter["feeder_gin"] = feeder_gin
    else:
        # Negative filter: No feeder selected → exclude feeder-required torches
        negative_filter["Feeder"] = True

    # Cooler logic with integrated awareness
    if cooler_gin:
        compatibility_filter["cooler_gin"] = cooler_gin
    elif not integrated_cooler:
        # Only exclude cooler-required torches if PS doesn't have integrated
        negative_filter["Cooler"] = True

    return await self._lucene_search(
        component_type="torch",
        user_message=user_message,
        limit=limit,
        offset=offset,
        min_score=min_score,
        compatibility_filter=compatibility_filter if compatibility_filter else None,
        negative_compatibility_filter=negative_filter if negative_filter else None
    )

    Pros: - Maintains generic architecture - Supports NOT EXISTS logic - Reusable for other components if needed

    Cons: - Requires signature change to generic method - More complex implementation

    async def search_torch_lucene(
    self,
    user_message: str,
    feeder_gin: Optional[str] = None,
    cooler_gin: Optional[str] = None,
    integrated_cooler: bool = False,
    limit: int = 10,
    offset: int = 0,
    min_score: float = 0.5
) -> SearchResults:
    # Step 1: Lucene search with positive compatibility only
    compatibility_filter = {}
    if feeder_gin:
        compatibility_filter["feeder_gin"] = feeder_gin
    if cooler_gin:
        compatibility_filter["cooler_gin"] = cooler_gin

    results = await self._lucene_search(
        component_type="torch",
        user_message=user_message,
        limit=limit * 2,  # Get more results to account for filtering
        offset=offset,
        min_score=min_score,
        compatibility_filter=compatibility_filter if compatibility_filter else None
    )

    # Step 2: Post-filter for NOT EXISTS logic
    # ... apply feeder/cooler NOT EXISTS rules ...
    filtered_products = self._apply_torch_not_exists_filter(
        results.products,
        has_feeder=bool(feeder_gin),
        has_cooler=bool(cooler_gin),
        integrated_cooler=integrated_cooler
    )

    return SearchResults(
        products=filtered_products[:limit],
        ...
    )

    Pros: - No changes to generic _lucene_search() - Clear separation of positive and negative filtering

    Cons: - Less efficient (fetch more, filter more) - Inconsistent with Cypher-based NOT EXISTS in traditional search
